const a0 = args[0], a1 = args[1]

//APPEND STRING WITH VALUE THROUGH REPLACE & ADDRESS SOME SPECIAL CASES
switch (a1) {
  case 'This armour wouldn&#39;t even protect you from a very weak blow' : gwc.output.replace ("This armour wouldn't even protect you from a very weak blow.", `${a1}. ${translator[a1]}`) ; break
